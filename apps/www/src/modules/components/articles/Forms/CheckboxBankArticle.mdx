interface CheckboxBankProps<T> {
  /** The currently selected choice(s) */
  value: T[]
  /** Possible choices */
  options: Array<{
    /** Visual label */
    label: string
    /** Value associated with the option */
    value: T
    /** Whether or not to enable the option */
    disabled?: boolean
  }>
  /** Name for the field. Used to generate unique id */
  name: string
  /** Whether or not to enable the _entire_ field */
  disabled: boolean
  /** Use alternate layout */
  inline: boolean
  /** Change handler */
  onChange: (values: T[]) => void
  /** Blur handler */
  onBlur: () => void
}
